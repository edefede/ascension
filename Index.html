Code Examples

// Crivello di Eratostene
func crivello(n) {
    array primi[n];

    // Inizializza tutti a 1 (primi)
    for (i = 0; i < n; i = i + 1) {
        primi[i] = 1;
    }

    // 0 e 1 non sono primi
    primi[0] = 0;
    primi[1] = 0;

    // Setaccia
    for (i = 2; i < n; i = i + 1) {
        if (primi[i] == 1) {
            // Elimina tutti i multipli di i
            for (j = i + i; j < n; j = j + i) {
                primi[j] = 0;
            }
        }
    }

    // Stampa i primi
    for (i = 0; i < n; i = i + 1) {
        if (primi[i] == 1) {
            print(i);
        }
    }
}

// Provalo con 100, poi 1000, 10000, 1000000!
crivello(100);
